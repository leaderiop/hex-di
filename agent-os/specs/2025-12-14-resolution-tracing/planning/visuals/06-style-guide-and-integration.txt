================================================================================
                    RESOLUTION TRACING - STYLE GUIDE & INTEGRATION
================================================================================

This document defines the style extensions needed for Resolution Tracing
and how they integrate with existing DevTools styles.

EXISTING STYLES TO REUSE (from styles.ts):
--------------------------------------------------------------------------------

From panelStyles:
  - container: Main panel wrapper
  - header: Panel title header
  - content: Content area padding

From sectionStyles:
  - header: Collapsible section header
  - headerHover: Hover state
  - title: Section title text
  - chevron / chevronExpanded: Expand indicator
  - content: Section content padding

From nodeStyles:
  - item: Node row container
  - name: Port name text
  - badge, badgeSingleton, badgeScoped, badgeRequest: Lifetime badges

From serviceListStyles:
  - searchContainer, searchInput, searchClearButton: Search UI
  - filterContainer, filterGroup, filterButton, filterButtonActive: Filters

From adapterStyles:
  - detailRow, detailLabel, detailValue: Detail rows

From emptyStyles:
  - container: Empty state message

UTILITY FUNCTIONS:
--------------------------------------------------------------------------------

/**
 * Get bar color style based on duration and threshold.
 */
export function getPerformanceBarStyle(
  durationMs: number,
  thresholdMs: number,
  isCacheHit: boolean
): CSSProperties {
  if (isCacheHit) {
    return timelineStyles.barCached;
  }
  if (durationMs >= thresholdMs) {
    return timelineStyles.barSlow;
  }
  if (durationMs >= thresholdMs * 0.5) {
    return timelineStyles.barMedium;
  }
  return timelineStyles.barFast;
}

/**
 * Get row style based on trace state.
 */
export function getTraceRowStyle(
  isExpanded: boolean,
  isSlow: boolean,
  isCacheHit: boolean
): CSSProperties {
  let style = { ...timelineStyles.row };

  if (isSlow) {
    style = { ...style, ...timelineStyles.rowSlow };
  } else if (isCacheHit) {
    style = { ...style, ...timelineStyles.rowCached };
  }

  if (isExpanded) {
    style = { ...style, ...timelineStyles.rowExpanded };
  }

  return style;
}

/**
 * Format duration for display.
 */
export function formatDuration(ms: number): string {
  if (ms < 0.1) {
    return "<0.1ms";
  }
  if (ms < 10) {
    return `${ms.toFixed(1)}ms`;
  }
  if (ms < 1000) {
    return `${Math.round(ms)}ms`;
  }
  return `${(ms / 1000).toFixed(2)}s`;
}
